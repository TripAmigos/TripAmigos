FIRST THINGS TO TRY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Sign Up (Create Account)
   ├─ Click "Get started"
   ├─ Enter email and password
   ├─ Complete 3 steps: Account → Profile → Preferences
   └─ You're logged in!

2. Create a Trip
   ├─ Click "+ New trip" button
   ├─ Enter trip name and dates
   ├─ Add attendees by email
   ├─ Set your preferences (optional)
   └─ Trip created!

3. View Dashboard
   ├─ See all your trips
   ├─ Check member responses
   ├─ Click trip to see details
   └─ Manage your plans

4. Log Out
   ├─ Click your avatar in top right
   ├─ Select "Sign out"
   └─ Redirected to homepage

Solution: rm -rf node_modules && npm install

Problem: Port 3000 in use
Solution: npm run dev -- -p 3001

Problem: Styling looks broken
Solution: rm -rf .next && npm run dev

Problem: Supabase connection error
Solution: Check .env.local has correct credentials
